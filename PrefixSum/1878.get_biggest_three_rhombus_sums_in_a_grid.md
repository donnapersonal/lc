# 1878.矩阵中最大的三个菱形和

题目链接：[1878.矩阵中最大的三个菱形和](https://leetcode.cn/problems/get-biggest-three-rhombus-sums-in-a-grid/)

## 题目大意

给你一个 m x n 的整数矩阵 grid 

菱形和 指的是 grid 中一个正菱形 边界 上的元素之和。本题中的菱形必须为正方形旋转45度，且四个角都在一个格子当中。下图是四个可行的菱形，每个菱形和应该包含的格子都用了相应颜色标注在图中

![alt text](https://github.com/donnapersonal/picx-images-hosting/raw/master/image.2kskcyqdgu.webp)

注意，菱形可以是一个面积为 0 的区域，如上图中右下角的紫色菱形所示

请你按照 降序 返回 grid 中三个最大的 互不相同的菱形和。如果不同的和少于三个，则将它们全部返回

![example](https://github.com/donnapersonal/picx-images-hosting/raw/master/image.1sfov8aous.webp)

限制:
- m == grid.length, n == grid[i].length
- 1 <= m, n <= 50
- 1 <= grid[i][j] <= 10^5

## 解题

如何枚举菱形？描述一个菱形需要几个变量？

如果确定了菱形的正中心 (i,j)，以及菱形顶点到正中心的距离 k，就确定了菱形的四个顶点的位置

如何快速计算菱形四条边的元素和？

可以用斜向前缀和加速：
- 定义 diagSum[i+1][j+1] 表示从矩阵最上边或最左边出发，向右下↘到 (i,j)，这条线段的元素和
- 定义 antiSum[i+1][j] 表示从矩阵最上边或最右边出发，向左下↙到 (i,j)，这条线段的元素和
  
为什么这里有一些 +1？原理在 前缀和 中讲了，类似一维数组的子数组和，即使子数组是前缀，我们仍然可以用两个前缀和之差算出子数组和，无需特判。

根据定义：
- 从 (x,y) 开始，向右下↘连续 k 个数的和为 diagSum[x+k][y+k]−diagSum[x][y]
- 从 (x,y) 开始，向左下↙连续 k 个数的和为 antiSum[x+k][y+1−k]−antiSum[x][y+1]

```js
/**
 * @param {number[][]} grid
 * @return {number[]}
 */
var getBiggestThree = function(grid) {
    const m = grid.length;
    const n = grid[0].length;

    const diag = Array.from({ length: m + 1 }, () => Array(n + 1).fill(0));
    const anti = Array.from({ length: m + 1 }, () => Array(n + 1).fill(0));

    for (let r = 0; r < m; r++) {
        for (let c = 0; c < n; c++) {
            diag[r + 1][c + 1] = diag[r][c] + grid[r][c];
            anti[r + 1][c] = anti[r][c + 1] + grid[r][c];
        }
    }

    function queryDiag(r, c, len) {
        return diag[r + len][c + len] - diag[r][c];
    }

    function queryAnti(r, c, len) {
        return anti[r + len][c + 1 - len] - anti[r][c + 1];
    }

    const top = [];

    function add(value) {
        if (top.includes(value)) return;

        top.push(value);
        top.sort((a, b) => b - a);

        if (top.length > 3) {
            top.pop();
        }
    }

    for (let r = 0; r < m; r++) {
        for (let c = 0; c < n; c++) {
            add(grid[r][c]);

            const maxRadius = Math.min(r, m - 1 - r, c, n - 1 - c);

            for (let k = 1; k <= maxRadius; k++) {
                const upperRight = queryDiag(r - k, c, k);
                const lowerLeft = queryDiag(r, c - k, k);
                const upperLeft = queryAnti(r - k + 1, c - 1, k - 1);
                const lowerRight = queryAnti(r, c + k, k + 1);

                add(upperRight + lowerLeft + upperLeft + lowerRight);
            }
        }
    }

    return top;
};
```

```python
from typing import List

class Solution:
    def getBiggestThree(self, grid: List[List[int]]) -> List[int]:
        m, n = len(grid), len(grid[0])

        # diag[r + 1][c + 1]: prefix sum on ↘ diagonal
        diag = [[0] * (n + 1) for _ in range(m + 1)]

        # anti[r + 1][c]: prefix sum on ↙ diagonal
        anti = [[0] * (n + 1) for _ in range(m + 1)]

        for r in range(m):
            for c in range(n):
                diag[r + 1][c + 1] = diag[r][c] + grid[r][c]
                anti[r + 1][c] = anti[r][c + 1] + grid[r][c]

        def query_diag(r: int, c: int, length: int) -> int:
            # Sum of length cells from (r, c) going ↘
            return diag[r + length][c + length] - diag[r][c]

        def query_anti(r: int, c: int, length: int) -> int:
            # Sum of length cells from (r, c) going ↙
            return anti[r + length][c + 1 - length] - anti[r][c + 1]

        top = []

        def add(value: int) -> None:
            if value in top:
                return
            top.append(value)
            top.sort(reverse=True)
            if len(top) > 3:
                top.pop()

        for r in range(m):
            for c in range(n):
                # A single cell is also a rhombus.
                add(grid[r][c])

                # radius = distance from center to each vertex
                max_radius = min(r, m - 1 - r, c, n - 1 - c)

                for radius in range(1, max_radius + 1):
                    k = radius

                    # Four edges of the rhombus.
                    upper_right = query_diag(r - k, c, k)
                    lower_left = query_diag(r, c - k, k)
                    upper_left = query_anti(r - k + 1, c - 1, k - 1)
                    lower_right = query_anti(r, c + k, k + 1)

                    rhombus_sum = upper_right + lower_left + upper_left + lower_right
                    add(rhombus_sum)

        return top
```
```java
import java.util.*;

class Solution {
    public int[] getBiggestThree(int[][] grid) {
        int m = grid.length;
        int n = grid[0].length;

        int[][] diag = new int[m + 1][n + 1];
        int[][] anti = new int[m + 1][n + 1];

        for (int r = 0; r < m; r++) {
            for (int c = 0; c < n; c++) {
                diag[r + 1][c + 1] = diag[r][c] + grid[r][c];
                anti[r + 1][c] = anti[r][c + 1] + grid[r][c];
            }
        }

        List<Integer> top = new ArrayList<>();

        for (int r = 0; r < m; r++) {
            for (int c = 0; c < n; c++) {
                add(top, grid[r][c]);

                int maxRadius = Math.min(
                    Math.min(r, m - 1 - r),
                    Math.min(c, n - 1 - c)
                );

                for (int k = 1; k <= maxRadius; k++) {
                    int upperRight = queryDiag(diag, r - k, c, k);
                    int lowerLeft = queryDiag(diag, r, c - k, k);
                    int upperLeft = queryAnti(anti, r - k + 1, c - 1, k - 1);
                    int lowerRight = queryAnti(anti, r, c + k, k + 1);

                    int sum = upperRight + lowerLeft + upperLeft + lowerRight;
                    add(top, sum);
                }
            }
        }

        int[] ans = new int[top.size()];
        for (int i = 0; i < top.size(); i++) {
            ans[i] = top.get(i);
        }
        return ans;
    }

    private int queryDiag(int[][] diag, int r, int c, int len) {
        return diag[r + len][c + len] - diag[r][c];
    }

    private int queryAnti(int[][] anti, int r, int c, int len) {
        return anti[r + len][c + 1 - len] - anti[r][c + 1];
    }

    private void add(List<Integer> top, int value) {
        if (top.contains(value)) {
            return;
        }

        top.add(value);
        top.sort(Collections.reverseOrder());

        if (top.size() > 3) {
            top.remove(top.size() - 1);
        }
    }
}
```

- 空间复杂度：`O(mn)`
