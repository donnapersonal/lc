# 372.超级次方

题目链接：[372.超级次方](https://leetcode.cn/problems/super-pow/)

## 题目大意

你的任务是计算 a^b 对 1337 取模，a 是一个正整数，b 是一个非常大的正整数且会以数组形式给出

```js
Example 1:
Input: a = 2, b = [3]
Output: 8

Example 2:
Input: a = 2, b = [1,0]
Output: 1024

Example 3:
Input: a = 1, b = [4,3,3,8,5,2]
Output: 1
```

限制：
- 1 <= a <= 2^31 - 1
- 1 <= b.length <= 2000
- 0 <= b[i] <= 9
- b does not contain leading zeros.

## 解题

根据题意，要求得的是 `a^b mod 1337` 的值，其中 `b` 是以数组形式给出。

是一道快速幂的题目（事实上也确实可以使用快速幂，但不是必须），但由于 `b` 是数组形式，因此还需要对其进行分解

可见，真正涉及计算次方的操作，所用到的次方都是一个 `10` 以内的数字，因此并非一定要使用`快速幂`

```js
// 快速幂
/**
 * @param {number} a
 * @param {number[]} b
 * @return {number}
 */
var superPow = function(a, b) {
    const MOD = 1337;

    const qpow = function(a, b) {
        let ans = 1;
        a %= MOD;
        while (b !== 0) {
            if ((b & 1) !== 0) ans = (ans * a) % MOD;
            a = (a * a) % MOD;
            b >>= 1;
        }
        return ans;
    };

    const dfs = function(a, b, u) {
        if (u === -1) return 1;
        return (qpow(dfs(a, b, u - 1), 10) * qpow(a, b[u])) % MOD;
    };

    return dfs(a, b, b.length - 1);
};

// 不使用快速幂
var superPow = function(a, b) {
    const MOD = 1337;

    const pow = function(a, k) {
        let ans = 1;
        a %= MOD;
        while (k-- > 0) {
            ans = (ans * a) % MOD;
        }
        return ans;
    };

    const dfs = function(a, b, u) {
        if (u === -1) return 1;
        return (pow(dfs(a, b, u - 1), 10) * pow(a, b[u])) % MOD;
    };

    return dfs(a, b, b.length - 1);
};
```
```python
# 快速幂
class Solution:
    MOD = 1337

    def superPow(self, a: int, b: List[int]) -> int:
        return self.dfs(a, b, len(b) - 1)

    def dfs(self, a: int, b: List[int], u: int) -> int:
        if u == -1:
            return 1
        return self.qpow(self.dfs(a, b, u - 1), 10) * self.qpow(a, b[u]) % self.MOD

    def qpow(self, a: int, b: int) -> int:
        res = 1
        a %= self.MOD
        while b != 0:
            if b & 1:
                res = res * a % self.MOD
            a = a * a % self.MOD
            b >>= 1
        return res

# 不使用快速幂
class Solution:
    def superPow(self, a: int, b: List[int]) -> int:
        self.MOD = 1337
        return self.dfs(a, b, len(b) - 1)

    def dfs(self, a: int, b: List[int], u: int) -> int:
        if u == -1:
            return 1
        return self.pow(self.dfs(a, b, u - 1), 10) * self.pow(a, b[u]) % self.MOD

    def pow(self, a: int, k: int) -> int:
        res = 1
        a %= self.MOD
        while k > 0:
            res = res * a % self.MOD
            k -= 1
        return res
```
```java
// 快速幂
class Solution {
    int MOD = 1337;
    public int superPow(int a, int[] b) {
        return dfs(a, b, b.length - 1);
    }
    int dfs(int a, int[] b, int u) {
        if (u == -1) return 1;
        return qpow(dfs(a, b, u - 1), 10) * qpow(a, b[u]) % MOD;
    }
    int qpow(int a, int b) {
        int ans = 1;
        a %= MOD;
        while (b != 0) {
            if ((b & 1) != 0) ans = ans * a % MOD;
            a = a * a % MOD;
            b >>= 1;
        }
        return ans;
    }
}

// 不使用快速幂
class Solution {
    int MOD = 1337;
    public int superPow(int a, int[] b) {
        return dfs(a, b, b.length - 1);
    }
    int dfs(int a, int[] b, int u) {
        if (u == -1) return 1;
        return pow(dfs(a, b, u - 1), 10) * pow(a, b[u]) % MOD;
    }
    int pow(int a, int b) {
        int ans = 1;
        a %= MOD;
        while (b-- > 0) ans = ans * a % MOD;
        return ans;
    }
}
```

- 时间复杂度：假设 `b` 数组所代表的数字为 `K`，使用快速幂的复杂度为 `O(logK)`，或者说是 `O(n∗log10)`，其中 `n` 为数组 `b` 的长度，数字 `10` 所代表的含义是计算一个次方为 `10` 以内的值；而不使用快速幂的复杂度为 `O(n∗10)`
- 空间复杂度：忽略递归带来的额外空间开销，复杂度为 `O(1)`
