# 3872.替换最多一个元素后的最长等差子数组

题目链接：[3872.替换最多一个元素后的最长等差子数组](https://leetcode.cn/problems/longest-arithmetic-sequence-after-changing-at-most-one-element/)

## 题目大意

给你一个整数数组 nums

如果子数组中相邻元素的差值是一个常数，那么这个子数组被称为 等差子数组

你可以将 nums 中的 最多 一个元素替换为任意一个 整数。然后，从 nums 中选择一个等差子数组

返回一个整数，该整数表示你可以选择的 最长 等差子数组的长度

子数组 是数组中一段连续的元素序列

```js
Example 1:
Input: nums = [9,7,5,10,1]
Output: 5
Explanation:
Replace nums[3] = 10 with 3. The array becomes [9, 7, 5, 3, 1].
Select the subarray [9, 7, 5, 3, 1], which is arithmetic because consecutive elements have a common difference of -2.

Example 2:
Input: nums = [1,2,6,7]
Output: 3
Explanation:
Replace nums[0] = 1 with -2. The array becomes [-2, 2, 6, 7].
Select the subarray [-2, 2, 6, 7], which is arithmetic because consecutive elements have a common difference of 4.
```

限制：
- 4 <= nums.length <= 10^5
- 1 <= nums[i] <= 10^5

## 解题

**分段扫描法**

适用场景：按照题目要求，数组会被分割成若干组，每一组的判断/处理逻辑是相同的

核心思想：
- 外层循环负责遍历组之前的准备工作（记录开始位置），和遍历组之后的统计工作（更新答案最大值）
- 内层循环负责遍历组，找出这一组最远在哪结束
  
这个写法的好处是，各个逻辑块分工明确，也不需要特判最后一组（易错点）。以我的经验，这个写法是所有写法中最不容易出 bug 的

对于本题，nums 包含若干段等差子数组。修改等差子数组的中间元素是无意义的（值没变），所以修改操作只会发生在等差子数组的端点旁边。用分组循环找到等差子数组的端点

```js
/**
 * @param {number[]} nums
 * @return {number}
 */
var longestArithmetic = function(nums) {
    const n = nums.length;

    if (n <= 2) {
        return n;
    }

    const calc = function(arr) {
        const len = arr.length;
        const pre = new Array(len).fill(0);

        pre[0] = 1;
        pre[1] = 2;

        for (let i = 2; i < len; i++) {
            if (arr[i - 2] + arr[i] === arr[i - 1] * 2) {
                pre[i] = pre[i - 1] + 1;
            } else {
                pre[i] = 2;
            }
        }

        return pre;
    };

    const pre = calc(nums);

    let res = Math.max(...pre) + 1;

    if (res >= n) {
        return n;
    }

    const reversed = [...nums].reverse();
    const suf = calc(reversed).reverse();

    for (let i = 1; i < n - 1; i++) {
        const d2 = nums[i + 1] - nums[i - 1];

        if (d2 % 2 !== 0) {
            continue;
        }

        const d = d2 / 2;

        const okLeft =
            i > 1 && nums[i - 1] - nums[i - 2] === d;

        const okRight =
            i + 2 < n && nums[i + 2] - nums[i + 1] === d;

        if (okLeft && okRight) {
            res = Math.max(res, pre[i - 1] + 1 + suf[i + 1]);
        } else if (okLeft) {
            res = Math.max(res, pre[i - 1] + 2);
        } else if (okRight) {
            res = Math.max(res, suf[i + 1] + 2);
        }
    }

    return Math.min(res, n);
};
```
```python
class Solution:
    def calc(self, nums: List[int]) -> List[int]:
        n = len(nums)
        pre = [0] * n
        pre[0] = 1
        pre[1] = 2
        for i in range(2, n):
            if nums[i - 2] + nums[i] == nums[i - 1] * 2:  # 三个数等差
                pre[i] = pre[i - 1] + 1
            else:
                pre[i] = 2
        return pre

    def longestArithmetic(self, nums: List[int]) -> int:
        n = len(nums)
        pre = self.calc(nums)
        res = max(pre) + 1
        if res >= n:  # 整个数组是等差的，或者修改端点元素后是等差的
            return n

        suf = self.calc(nums[::-1])[::-1]
        # 注意 max(pre) == max(suf)，无需重复计算

        for i in range(1, n - 1):
            # 把 nums[i] 改成 d2 / 2
            d2 = nums[i + 1] - nums[i - 1]
            if d2 % 2:  # d2 / 2 必须是整数
                continue

            ok_left = i > 1 and nums[i - 1] - nums[i - 2] == d2 // 2
            ok_right = i + 2 < n and nums[i + 2] - nums[i + 1] == d2 // 2

            if ok_left and ok_right:
                res = max(res, pre[i - 1] + 1 + suf[i + 1])
            elif ok_left:
                res = max(res, pre[i - 1] + 2)
            elif ok_right:
                res = max(res, suf[i + 1] + 2)

        return res
```
```java
import java.util.*;

class Solution {
    private int[] calc(int[] nums) {
        int n = nums.length;
        int[] pre = new int[n];

        if (n >= 1) {
            pre[0] = 1;
        }

        if (n >= 2) {
            pre[1] = 2;
        }

        for (int i = 2; i < n; i++) {
            // nums[i - 2], nums[i - 1], nums[i] are arithmetic
            if (nums[i - 2] + nums[i] == nums[i - 1] * 2) {
                pre[i] = pre[i - 1] + 1;
            } else {
                pre[i] = 2;
            }
        }

        return pre;
    }

    public int longestArithmetic(int[] nums) {
        int n = nums.length;

        if (n <= 2) {
            return n;
        }

        int[] pre = calc(nums);

        int res = 0;
        for (int len : pre) {
            res = Math.max(res, len);
        }

        res += 1;

        // The whole array is already arithmetic,
        // or we can modify one endpoint to make it arithmetic.
        if (res >= n) {
            return n;
        }

        int[] reversed = new int[n];

        for (int i = 0; i < n; i++) {
            reversed[i] = nums[n - 1 - i];
        }

        int[] suf = calc(reversed);

        // Reverse suf back
        for (int left = 0, right = n - 1; left < right; left++, right--) {
            int temp = suf[left];
            suf[left] = suf[right];
            suf[right] = temp;
        }

        for (int i = 1; i < n - 1; i++) {
            // Try to modify nums[i].
            // To make nums[i - 1], nums[i], nums[i + 1] arithmetic,
            // the new nums[i] should be (nums[i - 1] + nums[i + 1]) / 2.
            int d2 = nums[i + 1] - nums[i - 1];

            // The modified value must be an integer.
            if (d2 % 2 != 0) {
                continue;
            }

            int d = d2 / 2;

            boolean okLeft =
                i > 1 && nums[i - 1] - nums[i - 2] == d;

            boolean okRight =
                i + 2 < n && nums[i + 2] - nums[i + 1] == d;

            if (okLeft && okRight) {
                res = Math.max(res, pre[i - 1] + 1 + suf[i + 1]);
            } else if (okLeft) {
                res = Math.max(res, pre[i - 1] + 2);
            } else if (okRight) {
                res = Math.max(res, suf[i + 1] + 2);
            }
        }

        return Math.min(res, n);
    }
}
```

- 时间复杂度：`O(n)`，其中 `n` 是 `nums` 的长度，每个数至多被遍历两次
- 空间复杂度：`O(1)`
