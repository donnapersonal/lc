# 273.整数到英文单词

题目链接：[273.整数到英文单词](https://leetcode.com/problems/integer-to-english-words/)

## 题目大意

Convert a non-negative integer num to its English words representation

```js
Example 1:
Input: num = 123
Output: "One Hundred Twenty Three"

Example 2:
Input: num = 12345
Output: "Twelve Thousand Three Hundred Forty Five"

Example 3:
Input: num = 1234567
Output: "One Million Two Hundred Thirty Four Thousand Five Hundred Sixty Seven"
```

限制：
- 0 <= num <= 2^31 - 1

## 解题

```js
/**
 * @param {number} num
 * @return {string}
 */
var numberToWords = function(num) {
    if (num === 0) {
        return "Zero";
    }

    const lessThan20 = [
        "", "One", "Two", "Three", "Four", "Five", "Six", "Seven",
        "Eight", "Nine", "Ten", "Eleven", "Twelve", "Thirteen",
        "Fourteen", "Fifteen", "Sixteen", "Seventeen", "Eighteen",
        "Nineteen"
    ];

    const tens = [
        "", "", "Twenty", "Thirty", "Forty", "Fifty",
        "Sixty", "Seventy", "Eighty", "Ninety"
    ];

    const thousands = ["", "Thousand", "Million", "Billion"];

    const helper = (n) => {
        if (n === 0) {
            return "";
        }

        if (n < 20) {
            return lessThan20[n];
        }

        if (n < 100) {
            return tens[Math.floor(n / 10)] + (n % 10 !== 0 ? " " + helper(n % 10) : "");
        }

        return lessThan20[Math.floor(n / 100)] + " Hundred" + (n % 100 !== 0 ? " " + helper(n % 100) : "");
    };

    const res = [];
    let groupIndex = 0;

    while (num > 0) {
        const chunk = num % 1000;

        if (chunk !== 0) {
            let words = helper(chunk);

            if (thousands[groupIndex]) {
                words += " " + thousands[groupIndex];
            }

            res.push(words);
        }

        num = Math.floor(num / 1000);
        groupIndex++;
    }

    return res.reverse().join(" ");
};
```

```java
import java.util.*;

class Solution {
    private final String[] LESS_THAN_20 = {
        "", "One", "Two", "Three", "Four", "Five", "Six", "Seven",
        "Eight", "Nine", "Ten", "Eleven", "Twelve", "Thirteen",
        "Fourteen", "Fifteen", "Sixteen", "Seventeen", "Eighteen",
        "Nineteen"
    };

    private final String[] TENS = {
        "", "", "Twenty", "Thirty", "Forty", "Fifty",
        "Sixty", "Seventy", "Eighty", "Ninety"
    };

    private final String[] THOUSANDS = {
        "", "Thousand", "Million", "Billion"
    };

    public String numberToWords(int num) {
        if (num == 0) {
            return "Zero";
        }

        List<String> result = new ArrayList<>();
        int groupIndex = 0;

        while (num > 0) {
            int chunk = num % 1000;

            if (chunk != 0) {
                String words = helper(chunk);

                if (!THOUSANDS[groupIndex].isEmpty()) {
                    words += " " + THOUSANDS[groupIndex];
                }

                result.add(words);
            }

            num /= 1000;
            groupIndex++;
        }

        Collections.reverse(result);
        return String.join(" ", result);
    }

    private String helper(int num) {
        if (num == 0) {
            return "";
        }

        if (num < 20) {
            return LESS_THAN_20[num];
        }

        if (num < 100) {
            String result = TENS[num / 10];

            if (num % 10 != 0) {
                result += " " + helper(num % 10);
            }

            return result;
        }

        String result = LESS_THAN_20[num / 100] + " Hundred";

        if (num % 100 != 0) {
            result += " " + helper(num % 100);
        }

        return result;
    }
}
```

- Space Complexity: `O(d)`, for the result list.
