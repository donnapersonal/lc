# 3629.通过质数传送到达终点的最少跳跃次数

题目链接：[3629.通过质数传送到达终点的最少跳跃次数](https://leetcode.cn/problems/minimum-jumps-to-reach-end-via-prime-teleportation/)

## 题目大意

给你一个长度为 n 的整数数组 nums

你从下标 0 开始，目标是到达下标 n - 1

在任何下标 i 处，你可以执行以下操作之一：
- 移动到相邻格子：跳到下标 i + 1 或 i - 1，如果该下标在边界内
- 质数传送：如果 nums[i] 是一个质数 p，你可以立即跳到任何满足 nums[j] % p == 0 的下标 j 处，且下标 j != i 
  
返回到达下标 n - 1 所需的 最少 跳跃次数

质数 是一个大于 1 的自然数，只有两个因子，1 和它本身

```js
Example 1:
Input: nums = [1,2,4,6]
Output: 2
Explanation:
One optimal sequence of jumps is:
Start at index i = 0. Take an adjacent step to index 1.
At index i = 1, nums[1] = 2 is a prime number. Therefore, we teleport to index i = 3 as nums[3] = 6 is divisible by 2.
Thus, the answer is 2.

Example 2:
Input: nums = [2,3,4,7,9]
Output: 2
Explanation:
One optimal sequence of jumps is:
Start at index i = 0. Take an adjacent step to index i = 1.
At index i = 1, nums[1] = 3 is a prime number. Therefore, we teleport to index i = 4 since nums[4] = 9 is divisible by 3.
Thus, the answer is 2.

Example 3:
Input: nums = [4,6,5,8]
Output: 3
Explanation:
Since no teleportation is possible, we move through 0 → 1 → 2 → 3. Thus, the answer is 3.
```

限制：
- 1 <= n == nums.length <= 10^5
- 1 <= nums[i] <= 10^6

## 解题

从终点 n−1 跳到起点 0

跳跃规则反过来，从 i 跳到 nums[i] 的质因子 p 的下标

在执行 BFS 之前，对于 nums 中的每个数 x=nums[i]，如果 x 是质数，把 x 和下标 i 插入哈希表。哈希表的 key 是质数，value 是下标列表

```js
/**
 * @param {number[]} nums
 * @return {number}
 */
const MX = 1000001;

// spf[x] = smallest prime factor of x
const spf = new Int32Array(MX);

for (let i = 2; i < MX; i++) {
    if (spf[i] === 0) {
        for (let j = i; j < MX; j += i) {
            if (spf[j] === 0) {
                spf[j] = i;
            }
        }
    }
}

function isPrime(x) {
    return x >= 2 && spf[x] === x;
}

function getPrimeFactors(x) {
    const factors = [];

    while (x > 1) {
        const p = spf[x];
        factors.push(p);

        while (x % p === 0) {
            x = Math.floor(x / p);
        }
    }

    return factors;
}

var minJumps = function(nums) {
    const n = nums.length;

    // groups maps prime value p -> indices where nums[index] == p
    const groups = new Map();

    for (let i = 0; i < n; i++) {
        const x = nums[i];

        if (isPrime(x)) {
            if (!groups.has(x)) {
                groups.set(x, []);
            }
            groups.get(x).push(i);
        }
    }

    const visited = new Array(n).fill(false);
    visited[n - 1] = true;

    let q = [n - 1];
    let ans = 0;

    while (q.length > 0) {
        const nextQ = [];

        for (const i of q) {
            if (i === 0) {
                return ans;
            }

            if (i > 0 && !visited[i - 1]) {
                visited[i - 1] = true;
                nextQ.push(i - 1);
            }

            if (i < n - 1 && !visited[i + 1]) {
                visited[i + 1] = true;
                nextQ.push(i + 1);
            }

            const factors = getPrimeFactors(nums[i]);

            for (const p of factors) {
                const idx = groups.get(p);

                if (!idx) {
                    continue;
                }

                for (const j of idx) {
                    if (!visited[j]) {
                        visited[j] = true;
                        nextQ.push(j);
                    }
                }

                // Avoid scanning this prime group again
                groups.set(p, []);
            }
        }

        q = nextQ;
        ans++;
    }

    return -1;
};
```

```python
from typing import List
from collections import defaultdict
from itertools import count

MX = 1_000_001

# prime_factors[x] stores all unique prime factors of x
prime_factors = [[] for _ in range(MX)]

for i in range(2, MX):
    if not prime_factors[i]:  # i is prime
        for j in range(i, MX, i):
            prime_factors[j].append(i)


class Solution:
    def minJumps(self, nums: List[int]) -> int:
        n = len(nums)

        # groups[p] stores indices where nums[index] == p and p is prime
        groups = defaultdict(list)

        for i, x in enumerate(nums):
            if prime_factors[x] and prime_factors[x][0] == x:
                groups[x].append(i)

        vis = [False] * n
        vis[n - 1] = True

        q = [n - 1]

        for ans in count(0):
            tmp = q
            q = []

            for i in tmp:
                if i == 0:
                    return ans

                if i > 0 and not vis[i - 1]:
                    vis[i - 1] = True
                    q.append(i - 1)

                if i < n - 1 and not vis[i + 1]:
                    vis[i + 1] = True
                    q.append(i + 1)

                # Reverse jump:
                # If nums[i] has prime factor p,
                # then any index j with nums[j] == p can jump to i in the forward direction.
                for p in prime_factors[nums[i]]:
                    idx = groups[p]

                    for j in idx:
                        if not vis[j]:
                            vis[j] = True
                            q.append(j)

                    # Avoid scanning the same prime group again
                    idx.clear()
```

```java
import java.util.*;

class Solution {
    private static final int MX = 1_000_001;
    private static final int[] spf = new int[MX];

    static {
        for (int i = 2; i < MX; i++) {
            if (spf[i] == 0) {
                for (int j = i; j < MX; j += i) {
                    if (spf[j] == 0) {
                        spf[j] = i;
                    }
                }
            }
        }
    }

    public int minJumps(int[] nums) {
        int n = nums.length;

        // groups maps prime value p -> indices where nums[index] == p
        Map<Integer, List<Integer>> groups = new HashMap<>();

        for (int i = 0; i < n; i++) {
            int x = nums[i];

            if (isPrime(x)) {
                groups.computeIfAbsent(x, key -> new ArrayList<>()).add(i);
            }
        }

        boolean[] visited = new boolean[n];
        visited[n - 1] = true;

        Queue<Integer> q = new ArrayDeque<>();
        q.offer(n - 1);

        int ans = 0;

        while (!q.isEmpty()) {
            int size = q.size();

            for (int t = 0; t < size; t++) {
                int i = q.poll();

                if (i == 0) {
                    return ans;
                }

                if (i > 0 && !visited[i - 1]) {
                    visited[i - 1] = true;
                    q.offer(i - 1);
                }

                if (i < n - 1 && !visited[i + 1]) {
                    visited[i + 1] = true;
                    q.offer(i + 1);
                }

                List<Integer> factors = getPrimeFactors(nums[i]);

                for (int p : factors) {
                    List<Integer> idx = groups.get(p);

                    if (idx == null) {
                        continue;
                    }

                    for (int j : idx) {
                        if (!visited[j]) {
                            visited[j] = true;
                            q.offer(j);
                        }
                    }

                    // Avoid scanning the same prime group again
                    idx.clear();
                }
            }

            ans++;
        }

        return -1;
    }

    private boolean isPrime(int x) {
        return x >= 2 && spf[x] == x;
    }

    private List<Integer> getPrimeFactors(int x) {
        List<Integer> factors = new ArrayList<>();

        while (x > 1) {
            int p = spf[x];
            factors.add(p);

            while (x % p == 0) {
                x /= p;
            }
        }

        return factors;
    }
}
```

预处理的时间和空间不计入
- 时间复杂度：`O(nlogM)`，其中 `n` 是 `nums` 的长度，`M=max(nums)`
- 空间复杂度：`O(n)`
