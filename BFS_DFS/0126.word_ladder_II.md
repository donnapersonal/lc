# 126.单词接龙 II

题目链接：[126.单词接龙 II](https://leetcode.cn/problems/word-ladder-ii/)

## 题目大意

按字典 wordList 完成从单词 beginWord 到单词 endWord 转化，一个表示此过程的 转换序列 是形式上像 beginWord -> s1 -> s2 -> ... -> sk 这样的单词序列，并满足：
- 每对相邻的单词之间仅有单个字母不同
- 转换过程中的每个单词 si（1 <= i <= k）必须是字典 wordList 中的单词。注意，beginWord 不必是字典 wordList 中的单词
- sk == endWord
  
给你两个单词 beginWord 和 endWord ，以及一个字典 wordList 。请你找出并返回所有从 beginWord 到 endWord 的 最短转换序列 ，如果不存在这样的转换序列，返回一个空列表。每个序列都应该以单词列表 [beginWord, s1, s2, ..., sk] 的形式返回

```js
Example 1:
Input: beginWord = "hit", endWord = "cog", wordList = ["hot","dot","dog","lot","log","cog"]
Output: [["hit","hot","dot","dog","cog"],["hit","hot","lot","log","cog"]]
Explanation: There are 2 shortest transformation sequences:
"hit" -> "hot" -> "dot" -> "dog" -> "cog"
"hit" -> "hot" -> "lot" -> "log" -> "cog"

Example 2:
Input: beginWord = "hit", endWord = "cog", wordList = ["hot","dot","dog","lot","log"]
Output: []
Explanation: The endWord "cog" is not in wordList, therefore there is no valid transformation sequence.
```

限制:
- 1 <= beginWord.length <= 5
- endWord.length == beginWord.length
- 1 <= wordList.length <= 500
- wordList[i].length == beginWord.length
- beginWord, endWord, and wordList[i] consist of lowercase English letters.
- beginWord != endWord
- All the words in wordList are unique.
- The sum of all shortest transformation sequences does not exceed 10^5.

## 解题

**广度优先搜索 + 回溯**


本题要求的是最短转换序列，看到最短首先想到的就是广度优先搜索。但是本题没有给出显示的图结构，根据单词转换规则：把每个单词都抽象为一个顶点，如果两个单词可以只改变一个字母进行转换，那么说明它们之间有一条双向边。因此我们只需要把满足转换条件的点相连，就形成了一张图

从一个单词出发，修改每一位字符，将它修改成为 ‘a’ 到 ‘z’ 中的所有字符，看看修改以后是不是在题目中给出的单词列表中

有一些边的关系，由于不是最短路径上的边，不可以被记录下来。为此，我们为扩展出的单词记录附加的属性：层数。即下面代码中的 steps。如果当前的单词扩散出去得到的单词的层数在以前出现过，则不应该记录这样的边的关系

```js
/**
 * @param {string} beginWord
 * @param {string} endWord
 * @param {string[]} wordList
 * @return {string[][]}
 */
var findLadders = function(beginWord, endWord, wordList) {
    const res = [];
    const wordSet = new Set(wordList);

    if (!wordSet.has(endWord)) {
        return res;
    }

    wordSet.delete(beginWord);

    const parents = new Map(); // child -> list of parents
    const steps = new Map();

    steps.set(beginWord, 0);

    const queue = [beginWord];
    let head = 0;

    let step = 1;
    let found = false;
    const wordLen = beginWord.length;

    function addParent(child, parent) {
        if (!parents.has(child)) {
            parents.set(child, []);
        }
        parents.get(child).push(parent);
    }

    while (head < queue.length && !found) {
        const size = queue.length - head;

        for (let t = 0; t < size; t++) {
            const cur = queue[head++];
            const chars = cur.split("");

            for (let i = 0; i < wordLen; i++) {
                const original = chars[i];

                for (let code = 97; code <= 122; code++) {
                    chars[i] = String.fromCharCode(code);
                    const nxt = chars.join("");

                    if (steps.has(nxt) && steps.get(nxt) === step) {
                        addParent(nxt, cur);
                    }

                    if (!wordSet.has(nxt)) {
                        continue;
                    }

                    wordSet.delete(nxt);
                    queue.push(nxt);

                    addParent(nxt, cur);
                    steps.set(nxt, step);

                    if (nxt === endWord) {
                        found = true;
                    }
                }

                chars[i] = original;
            }
        }

        step++;
    }

    if (!found) {
        return res;
    }

    const path = [endWord];

    function backtrack(word) {
        if (word === beginWord) {
            res.push([...path].reverse());
            return;
        }

        const list = parents.get(word) || [];

        for (const parent of list) {
            path.push(parent);
            backtrack(parent);
            path.pop();
        }
    }

    backtrack(endWord);
    return res;
};
```

```python
from collections import defaultdict, deque
from typing import List

class Solution:
    def findLadders(self, beginWord: str, endWord: str, wordList: List[str]) -> List[List[str]]:
        res = []
        word_set = set(wordList)

        if endWord not in word_set:
            return res

        word_set.discard(beginWord)

        parents = defaultdict(list)   # child -> list of parents
        steps = {beginWord: 0}

        queue = deque([beginWord])
        step = 1
        found = False
        word_len = len(beginWord)

        while queue and not found:
            size = len(queue)

            for _ in range(size):
                cur = queue.popleft()
                chars = list(cur)

                for i in range(word_len):
                    original = chars[i]

                    for c in "abcdefghijklmnopqrstuvwxyz":
                        chars[i] = c
                        nxt = "".join(chars)

                        if nxt in steps and steps[nxt] == step:
                            parents[nxt].append(cur)

                        if nxt not in word_set:
                            continue

                        word_set.remove(nxt)
                        queue.append(nxt)

                        parents[nxt].append(cur)
                        steps[nxt] = step

                        if nxt == endWord:
                            found = True

                    chars[i] = original

            step += 1

        if not found:
            return res

        path = [endWord]

        def backtrack(word):
            if word == beginWord:
                res.append(path[::-1])
                return

            for parent in parents[word]:
                path.append(parent)
                backtrack(parent)
                path.pop()

        backtrack(endWord)
        return res
```

```java
import java.util.*;

class Solution {
    public List<List<String>> findLadders(String beginWord, String endWord, List<String> wordList) {
        List<List<String>> res = new ArrayList<>();

        Set<String> wordSet = new HashSet<>(wordList);

        if (!wordSet.contains(endWord)) {
            return res;
        }

        wordSet.remove(beginWord);

        Map<String, List<String>> parents = new HashMap<>();
        Map<String, Integer> steps = new HashMap<>();

        steps.put(beginWord, 0);

        Queue<String> queue = new ArrayDeque<>();
        queue.offer(beginWord);

        int step = 1;
        boolean found = false;
        int wordLen = beginWord.length();

        while (!queue.isEmpty() && !found) {
            int size = queue.size();

            for (int t = 0; t < size; t++) {
                String cur = queue.poll();
                char[] chars = cur.toCharArray();

                for (int i = 0; i < wordLen; i++) {
                    char original = chars[i];

                    for (char c = 'a'; c <= 'z'; c++) {
                        chars[i] = c;
                        String nxt = new String(chars);

                        if (steps.containsKey(nxt) && steps.get(nxt) == step) {
                            parents.computeIfAbsent(nxt, key -> new ArrayList<>()).add(cur);
                        }

                        if (!wordSet.contains(nxt)) {
                            continue;
                        }

                        wordSet.remove(nxt);
                        queue.offer(nxt);

                        parents.computeIfAbsent(nxt, key -> new ArrayList<>()).add(cur);
                        steps.put(nxt, step);

                        if (nxt.equals(endWord)) {
                            found = true;
                        }
                    }

                    chars[i] = original;
                }
            }

            step++;
        }

        if (!found) {
            return res;
        }

        LinkedList<String> path = new LinkedList<>();
        path.add(endWord);

        backtrack(endWord, beginWord, parents, path, res);

        return res;
    }

    private void backtrack(
        String word,
        String beginWord,
        Map<String, List<String>> parents,
        LinkedList<String> path,
        List<List<String>> res
    ) {
        if (word.equals(beginWord)) {
            List<String> onePath = new ArrayList<>(path);
            Collections.reverse(onePath);
            res.add(onePath);
            return;
        }

        for (String parent : parents.getOrDefault(word, new ArrayList<>())) {
            path.add(parent);
            backtrack(parent, beginWord, parents, path, res);
            path.removeLast();
        }
    }
}
```
