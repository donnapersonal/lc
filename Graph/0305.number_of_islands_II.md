# 305.岛屿数量 II

题目链接：[305.岛屿数量 II](https://leetcode.cn/problems/number-of-islands-ii/)

## 题目大意

给你一个大小为 m x n 的二维二进制网格 grid 。网格表示一个地图，其中，0 表示水，1 表示陆地。最初，grid 中的所有单元格都是水单元格（即，所有单元格都是 0）

可以通过执行 addLand 操作，将某个位置的水转换成陆地。给你一个数组 positions ，其中 positions[i] = [ri, ci] 是要执行第 i 次操作的位置 (ri, ci) 

返回一个整数数组 answer ，其中 answer[i] 是将单元格 (ri, ci) 转换为陆地后，地图中岛屿的数量

岛屿 的定义是被「水」包围的「陆地」，通过水平方向或者垂直方向上相邻的陆地连接而成。你可以假设地图网格的四边均被无边无际的「水」所包围

![example](https://github.com/donnapersonal/picx-images-hosting/raw/master/image.7w7grn1tam.webp)

限制：
- 1 <= m, n, positions.length <= 10^4
- 1 <= m * n <= 10^4
- positions[i].length == 2
- 0 <= ri < m
- 0 <= ci < n

## 解题

这题是一个动态加陆地的问题

一开始整个网格都是水，每次 positions 里给一个坐标，把这个位置变成陆地，然后返回当前岛屿数量

如果每次加完陆地都重新 DFS/BFS 统计岛屿会很慢，所以用 并查集 Union-Find 维护连通块数量

**关键逻辑**

每新增一个陆地：
- 如果这个位置之前已经是陆地，岛屿数量不变
- 如果是新陆地，先让岛屿数量 +1
- 检查上下左右 4 个邻居
- 如果邻居也是陆地，就把当前陆地和邻居合并
- 每成功合并两个不同岛屿，岛屿数量 -1
- 把当前岛屿数量加入结果
  
**为什么要用并查集？**
- 因为`岛屿本质是连通块`
- 每次新增陆地时，它可能：
  - 单独形成一个新岛
  - 和旁边一个岛连起来
  - 同时把多个岛连成一个岛
- 并查集非常适合处理这种“动态合并连通块”的问题

/**
 * @param {number} m
 * @param {number} n
 * @param {number[][]} positions
 * @return {number[]}
 */
class UnionFind {
    constructor(n) {
        this.parent = new Array(n);
        this.size = new Array(n).fill(1);
        this.count = 0;

        for (let i = 0; i < n; i++) {
            this.parent[i] = i;
        }
    }

    find(x) {
        if (this.parent[x] !== x) {
            this.parent[x] = this.find(this.parent[x]);
        }
        return this.parent[x];
    }

    union(x, y) {
        let rootX = this.find(x);
        let rootY = this.find(y);

        if (rootX === rootY) {
            return false;
        }

        if (this.size[rootX] > this.size[rootY]) {
            [rootX, rootY] = [rootY, rootX];
        }

        this.parent[rootX] = rootY;
        this.size[rootY] += this.size[rootX];
        this.count--;

        return true;
    }
}

/**
 * @param {number} m
 * @param {number} n
 * @param {number[][]} positions
 * @return {number[]}
 */
var numIslands2 = function(m, n, positions) {
    const uf = new UnionFind(m * n);
    const result = [];
    const seen = new Set();

    const directions = [
        [-1, 0],
        [1, 0],
        [0, -1],
        [0, 1]
    ];

    for (const [r, c] of positions) {
        const index = r * n + c;

        if (seen.has(index)) {
            result.push(uf.count);
            continue;
        }

        seen.add(index);
        uf.count++;

        for (const [dr, dc] of directions) {
            const nr = r + dr;
            const nc = c + dc;

            if (nr >= 0 && nr < m && nc >= 0 && nc < n) {
                const neighbor = nr * n + nc;

                if (seen.has(neighbor)) {
                    uf.union(index, neighbor);
                }
            }
        }

        result.push(uf.count);
    }

    return result;
};

```python
class UnionFind:
    def __init__(self, n):
        self.parent = [x for x in range(n)]
        self.size = [1 for x in range(n)]
        self.part = 0

    def Find(self, x: int) -> int:
        if self.parent[x] != x:
            self.parent[x] = self.Find(self.parent[x])
        return self.parent[x]

    def Union(self, x: int, y: int) -> bool:
        root_x = self.Find(x)
        root_y = self.Find(y)

        if root_x == root_y:
            return False

        if self.size[root_x] > self.size[root_y]:
            root_x, root_y = root_y, root_x

        self.parent[root_x] = root_y
        self.size[root_y] += self.size[root_x]
        self.part -= 1
        return True

class Solution:
    def numIslands2(self, m: int, n: int, positions: List[List[int]]) -> List[int]:
        uf = UnionFind(m * n)
        res = []
        seen = set()

        for r, c in positions:
            idx = r * n + c

            if idx in seen:
                res.append(uf.part)
                continue

            seen.add(idx)
            uf.part += 1

            for nr, nc in ((r - 1, c), (r + 1, c), (r, c - 1), (r, c + 1)):
                if 0 <= nr < m and 0 <= nc < n:
                    nei = nr * n + nc
                    if nei in seen:
                        uf.Union(idx, nei)

            res.append(uf.part)

        return res
```
```java
import java.util.*;

class UnionFind {
    int[] parent;
    int[] size;
    int count;

    public UnionFind(int n) {
        parent = new int[n];
        size = new int[n];
        count = 0;

        for (int i = 0; i < n; i++) {
            parent[i] = i;
            size[i] = 1;
        }
    }

    public int find(int x) {
        if (parent[x] != x) {
            parent[x] = find(parent[x]);
        }
        return parent[x];
    }

    public boolean union(int x, int y) {
        int rootX = find(x);
        int rootY = find(y);

        if (rootX == rootY) {
            return false;
        }

        if (size[rootX] > size[rootY]) {
            int temp = rootX;
            rootX = rootY;
            rootY = temp;
        }

        parent[rootX] = rootY;
        size[rootY] += size[rootX];
        count--;

        return true;
    }
}

class Solution {
    public List<Integer> numIslands2(int m, int n, int[][] positions) {
        UnionFind uf = new UnionFind(m * n);
        List<Integer> result = new ArrayList<>();
        Set<Integer> seen = new HashSet<>();

        int[][] directions = {
            {-1, 0},
            {1, 0},
            {0, -1},
            {0, 1}
        };

        for (int[] pos : positions) {
            int r = pos[0];
            int c = pos[1];

            int index = r * n + c;

            if (seen.contains(index)) {
                result.add(uf.count);
                continue;
            }

            seen.add(index);
            uf.count++;

            for (int[] dir : directions) {
                int nr = r + dir[0];
                int nc = c + dir[1];

                if (nr >= 0 && nr < m && nc >= 0 && nc < n) {
                    int neighbor = nr * n + nc;

                    if (seen.contains(neighbor)) {
                        uf.union(index, neighbor);
                    }
                }
            }

            result.add(uf.count);
        }

        return result;
    }
}
```

设：P = len(positions)，N = m * n，每次添加一个位置，最多检查 4 个邻居， 每次 find/union 的均摊复杂度接近：O(α(N))， 其中 α 是反阿克曼函数，几乎可以看作常数

- 时间复杂度：`O(P * α(mn))`，面试里可以说接近：O(P)
- 空间复杂度：`O(mn)`
  - parent 数组大小是 m * n
  - size 数组大小是 m * n
  - seen 最多存 m * n 个点
