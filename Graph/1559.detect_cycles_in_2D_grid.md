# 1559.二维网格图中探测环

题目链接：[1559.二维网格图中探测环](https://leetcode.cn/problems/detect-cycles-in-2d-grid/)

## 题目大意

给你一个二维字符网格数组 grid ，大小为 m x n ，你需要检查 grid 中是否存在 相同值 形成的环

一个环是一条开始和结束于同一个格子的长度 大于等于 4 的路径。对于一个给定的格子，你可以移动到它上、下、左、右四个方向相邻的格子之一，可以移动的前提是这两个格子有 相同的值 

同时，你也不能回到上一次移动时所在的格子。比方说，环  (1, 1) -> (1, 2) -> (1, 1) 是不合法的，因为从 (1, 2) 移动到 (1, 1) 回到了上一次移动时的格子

如果 grid 中有相同值形成的环，请你返回 true ，否则返回 false

- 1 <= m, n <= 500
- grid consists only of lowercase English letters.

## 解题

在 `DFS` 的过程中，如果发现了一个之前访问过的同值格子，那么：
- 如果从 (x,y) 移动一步到 (i,j)，再从 (i,j) 移动一步到 (x,y)，这的确是个环，但长度只有 2，不满足要求。所以规定，不能立刻回头。在 DFS 的过程中，额外传入上一步的位置 (px,py)，并禁止从当前位置移动到上一步的位置
- 如果我们从 (x,y) 移动一步到 (i,j)，且 (i,j) 不等于 (px,py)，且 (i,j) 之前访问过，那么就找到了一个长度至少为 4 的环
  
> 有没有可能，环长是 3？
> - 对于（四连通的）网格图，这是不可能的。把网格图看成国际象棋棋盘，每走一步，所处格子的颜色切换成另一种颜色。对于一个环，从环上的一点出发，只有走偶数步后，所处格子的颜色才和起始格子的颜色相同。所以对于（四连通的）网格图，环长一定是偶数。只要不立刻回头，环长至少为 4

```js
/**
 * @param {character[][]} grid
 * @return {boolean}
 */
var containsCycle = function(grid) {
    const m = grid.length;
    const n = grid[0].length;

    const visited = Array.from({ length: m }, () => Array(n).fill(false));

    function dfs(x, y, px, py) {
        visited[x][y] = true;

        const dirs = [
            [0, -1],
            [0, 1],
            [-1, 0],
            [1, 0]
        ];

        for (const [dx, dy] of dirs) {
            const nx = x + dx;
            const ny = y + dy;

            // Skip the parent cell
            if (nx === px && ny === py) {
                continue;
            }

            // Check boundary
            if (nx < 0 || nx >= m || ny < 0 || ny >= n) {
                continue;
            }

            // Only move to cells with the same character
            if (grid[nx][ny] !== grid[x][y]) {
                continue;
            }

            // If this same-character neighbor was visited before,
            // and it is not the parent, then we found a cycle.
            if (visited[nx][ny]) {
                return true;
            }

            if (dfs(nx, ny, x, y)) {
                return true;
            }
        }

        return false;
    }

    for (let i = 0; i < m; i++) {
        for (let j = 0; j < n; j++) {
            if (!visited[i][j]) {
                if (dfs(i, j, -1, -1)) {
                    return true;
                }
            }
        }
    }

    return false;
};
```

class Solution:
    def containsCycle(self, grid: List[List[str]]) -> bool:
        m, n = len(grid), len(grid[0])
        visited = [[False] * n for _ in range(m)]

        def dfs(x: int, y: int, px: int, py: int) -> bool:
            visited[x][y] = True
            for i, j in (x, y - 1), (x, y + 1), (x - 1, y), (x + 1, y):  # 枚举移动方向
                if ((i != px or j != py) and  # (i, j) 不是上一步的格子 (px, py)
                    0 <= i < m and 0 <= j < n and  # (i, j) 没有出界
                    grid[i][j] == grid[x][y] and  # (i, j) 和 (x, y) 的格子值相同
                    (visited[i][j] or dfs(i, j, x, y))):  # 如果之前访问过 (i, j)，那么找到了环，否则继续递归找
                    return True
            return False

        for i in range(m):
            for j in range(n):
                if not visited[i][j] and dfs(i, j, -1, -1):
                    return True
        return False

from typing import List

class Solution:
    def containsCycle(self, grid: List[List[str]]) -> bool:
        m, n = len(grid), len(grid[0])
        visited = [[False] * n for _ in range(m)]

        directions = [(0, 1), (0, -1), (1, 0), (-1, 0)]

        def dfs(x: int, y: int, parent_x: int, parent_y: int) -> bool:
            visited[x][y] = True

            for dx, dy in directions:
                nx = x + dx
                ny = y + dy

                if nx < 0 or nx >= m or ny < 0 or ny >= n:
                    continue

                if grid[nx][ny] != grid[x][y]:
                    continue

                if nx == parent_x and ny == parent_y:
                    continue

                if visited[nx][ny]:
                    return True

                if dfs(nx, ny, x, y):
                    return True

            return False

        for i in range(m):
            for j in range(n):
                if not visited[i][j]:
                    if dfs(i, j, -1, -1):
                        return True
        return False

```java
class Solution {
    private int m;
    private int n;
    private char[][] grid;
    private boolean[][] visited;

    public boolean containsCycle(char[][] grid) {
        this.grid = grid;
        this.m = grid.length;
        this.n = grid[0].length;
        this.visited = new boolean[m][n];

        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (!visited[i][j]) {
                    if (dfs(i, j, -1, -1)) {
                        return true;
                    }
                }
            }
        }

        return false;
    }

    private boolean dfs(int x, int y, int px, int py) {
        visited[x][y] = true;

        int[][] dirs = {
            {0, -1},
            {0, 1},
            {-1, 0},
            {1, 0}
        };

        for (int[] dir : dirs) {
            int nx = x + dir[0];
            int ny = y + dir[1];

            // Skip the parent cell
            if (nx == px && ny == py) {
                continue;
            }

            // Check boundary
            if (nx < 0 || nx >= m || ny < 0 || ny >= n) {
                continue;
            }

            // Only move to cells with the same character
            if (grid[nx][ny] != grid[x][y]) {
                continue;
            }

            // If this same-character neighbor was visited before,
            // and it is not the parent, then we found a cycle.
            if (visited[nx][ny]) {
                return true;
            }

            if (dfs(nx, ny, x, y)) {
                return true;
            }
        }

        return false;
    }
}
```

- 时间复杂度：`O(mn)`，其中 `m` 和 `n` 分别是 `grid` 的行数和列数
- 空间复杂度：`O(mn)`
