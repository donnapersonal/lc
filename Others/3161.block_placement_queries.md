# 3161.物块放置查询

题目链接：[3161.物块放置查询](https://leetcode.cn/problems/block-placement-queries/)

## 题目大意

有一条无限长的数轴，原点在 0 处，沿着 x 轴 正 方向无限延伸

给你一个二维数组 queries ，它包含两种操作：
- 操作类型 1 ：queries[i] = [1, x] 。在距离原点 x 处建一个障碍物。数据保证当操作执行的时候，位置 x 处 没有 任何障碍物
- 操作类型 2 ：queries[i] = [2, x, sz] 。判断在数轴范围 [0, x] 内是否可以放置一个长度为 sz 的物块，这个物块需要 完全 放置在范围 [0, x] 内。如果物块与任何障碍物有重合，那么这个物块 不能 被放置，但物块可以与障碍物刚好接触。注意，你只是进行查询，并 不是 真的放置这个物块。每个查询都是相互独立的
  
请返回一个 boolean 数组results ，如果第 i 个操作类型 2 的操作你可以放置物块，那么 results[i] 为 true ，否则为 false 

![example](https://github.com/donnapersonal/picx-images-hosting/raw/master/image.96adzr7j3m.webp)

限制：
- 1 <= queries.length <= 15 * 10^4
- 2 <= queries[i].length <= 3
- 1 <= queries[i][0] <= 2
- 1 <= x, sz <= min(5 * 10^4, 3 * queries.length)
- The input is generated such that for queries of type 1, no obstacle exists at distance x when the query is asked.
- The input is generated such that there is at least one query of type 2.

## 解题

**正序回答询问+有序集合+线段树**

```js
/**
 * @param {number[][]} queries
 * @return {boolean[]}
 */
var getResults = function(queries) {
    let m = 0;
    for (const q of queries) {
        m = Math.max(m, q[1]);
    }
    m++;

    const seg = new Array(4 * (m + 1)).fill(0);

    function update(node, l, r, idx, val) {
        if (l === r) {
            seg[node] = val;
            return;
        }

        const mid = Math.floor((l + r) / 2);

        if (idx <= mid) {
            update(node * 2, l, mid, idx, val);
        } else {
            update(node * 2 + 1, mid + 1, r, idx, val);
        }

        seg[node] = Math.max(seg[node * 2], seg[node * 2 + 1]);
    }

    function query(node, l, r, R) {
        if (r <= R) {
            return seg[node];
        }

        const mid = Math.floor((l + r) / 2);

        if (R <= mid) {
            return query(node * 2, l, mid, R);
        }

        return Math.max(seg[node * 2], query(node * 2 + 1, mid + 1, r, R));
    }

    function lowerBound(arr, target) {
        let left = 0;
        let right = arr.length;

        while (left < right) {
            const mid = Math.floor((left + right) / 2);
            if (arr[mid] < target) {
                left = mid + 1;
            } else {
                right = mid;
            }
        }

        return left;
    }

    const obstacles = [0, m];
    const res = [];

    for (const q of queries) {
        const x = q[1];
        const i = lowerBound(obstacles, x);
        const pre = obstacles[i - 1];

        if (q[0] === 1) {
            const nxt = obstacles[i];

            obstacles.splice(i, 0, x);

            update(1, 0, m, x, x - pre);
            update(1, 0, m, nxt, nxt - x);
        } else {
            const size = q[2];
            const maxGap = Math.max(query(1, 0, m, pre), x - pre);
            res.push(maxGap >= size);
        }
    }

    return res;
};
```

```python
class Solution:
    def getResults(self, queries: List[List[int]]) -> List[bool]:
        m = max(q[1] for q in queries) + 1
        t = [0] * (2 << m.bit_length())

        def update(o: int, l: int, r: int, i: int, val: int) -> None:
            if l == r:
                t[o] = val
                return
            m = (l + r) // 2
            if i <= m:
                update(o * 2, l, m, i, val)
            else:
                update(o * 2 + 1, m + 1, r, i, val)
            t[o] = max(t[o * 2], t[o * 2 + 1])

        def query(o: int, l: int, r: int, R: int) -> int:
            if r <= R:
                return t[o]
            m = (l + r) // 2
            if R <= m:
                return query(o * 2, l, m, R)
            return max(t[o * 2], query(o * 2 + 1, m + 1, r, R))

        sl = SortedList([0, m])
        res = []

        for q in queries:
            x = q[1]
            i = sl.bisect_left(x)
            pre = sl[i - 1]

            if q[0] == 1:
                nxt = sl[i]
                sl.add(x)
                update(1, 0, m, x, x - pre)
                update(1, 0, m, nxt, nxt - x)
            else:
                max_gap = max(query(1, 0, m, pre), x - pre)
                res.append(max_gap >= q[2])

        return res
```

```java
import java.util.*;

class Solution {
    private int[] seg;
    private int m;

    public List<Boolean> getResults(int[][] queries) {
        m = 0;
        for (int[] q : queries) {
            m = Math.max(m, q[1]);
        }
        m++;

        seg = new int[4 * (m + 1)];

        TreeSet<Integer> obstacles = new TreeSet<>();
        obstacles.add(0);
        obstacles.add(m);

        List<Boolean> res = new ArrayList<>();

        for (int[] q : queries) {
            int x = q[1];

            Integer pre = obstacles.lower(x);
            if (pre == null) {
                pre = 0;
            }

            if (q[0] == 1) {
                Integer nxt = obstacles.ceiling(x);

                obstacles.add(x);

                update(1, 0, m, x, x - pre);
                update(1, 0, m, nxt, nxt - x);
            } else {
                int size = q[2];
                int maxGap = Math.max(query(1, 0, m, pre), x - pre);
                res.add(maxGap >= size);
            }
        }

        return res;
    }

    private void update(int node, int l, int r, int idx, int val) {
        if (l == r) {
            seg[node] = val;
            return;
        }

        int mid = l + (r - l) / 2;

        if (idx <= mid) {
            update(node * 2, l, mid, idx, val);
        } else {
            update(node * 2 + 1, mid + 1, r, idx, val);
        }

        seg[node] = Math.max(seg[node * 2], seg[node * 2 + 1]);
    }

    private int query(int node, int l, int r, int R) {
        if (r <= R) {
            return seg[node];
        }

        int mid = l + (r - l) / 2;

        if (R <= mid) {
            return query(node * 2, l, mid, R);
        }

        return Math.max(seg[node * 2], query(node * 2 + 1, mid + 1, r, R));
    }
}
```

- 空间复杂度：`O(U)`
