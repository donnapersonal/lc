# 3721.最长平衡子数组 II

题目链接：[3721.最长平衡子数组 II](https://leetcode.cn/problems/longest-balanced-subarray-ii/)

## 题目大意

给你一个整数数组 nums。

如果子数组中 不同偶数 的数量等于 不同奇数 的数量，则称该 子数组 是 平衡的 

返回 最长 平衡子数组的长度

子数组 是数组中连续且 非空 的一段元素序列

```js
Example 1:
Input: nums = [2,5,4,3]
Output: 4
Explanation:
The longest balanced subarray is [2, 5, 4, 3].
It has 2 distinct even numbers [2, 4] and 2 distinct odd numbers [5, 3]. Thus, the answer is 4.

Example 2:
Input: nums = [3,2,2,5,4]
Output: 5
Explanation:
The longest balanced subarray is [3, 2, 2, 5, 4].
It has 2 distinct even numbers [2, 4] and 2 distinct odd numbers [3, 5]. Thus, the answer is 5.

Example 3:
Input: nums = [1,2,3,2]
Output: 3
Explanation:
The longest balanced subarray is [2, 3, 2].
It has 1 distinct even number [2] and 1 distinct odd number [3]. Thus, the answer is 3.
```

限制：
- 1 <= nums.length <= 10^5
- 1 <= nums[i] <= 10^5

## 解题

**Lazy 线段树**

由于 a 中元素只有 −1,0,1，所以 sum 数组相邻元素之差 ≤1。根据离散介值定理，设 min 和 max 分别为区间的最小值和最大值，只要 sum[i] 在 [min,max] 范围中，区间就一定存在等于 sum[i] 的数

用 Lazy 线段树维护区间最小值、区间最大值、区间加的 Lazy tag

```js
/**
 * @param {number[]} nums
 * @return {number}
 */
class Node {
    constructor() {
        this.min = 0;
        this.max = 0;
        this.lazy = 0;
    }
}

class LazySegmentTree {
    constructor(n) {
        this.n = n;
        this.tree = Array.from({ length: 4 * n }, () => new Node());
    }

    apply(node, delta) {
        this.tree[node].min += delta;
        this.tree[node].max += delta;
        this.tree[node].lazy += delta;
    }

    pushDown(node) {
        const lazy = this.tree[node].lazy;
        if (lazy === 0) return;

        this.apply(node * 2, lazy);
        this.apply(node * 2 + 1, lazy);
        this.tree[node].lazy = 0;
    }

    pushUp(node) {
        this.tree[node].min = Math.min(
            this.tree[node * 2].min,
            this.tree[node * 2 + 1].min
        );

        this.tree[node].max = Math.max(
            this.tree[node * 2].max,
            this.tree[node * 2 + 1].max
        );
    }

    updateRange(node, l, r, ql, qr, delta) {
        if (ql <= l && r <= qr) {
            this.apply(node, delta);
            return;
        }

        this.pushDown(node);

        const mid = Math.floor((l + r) / 2);

        if (ql <= mid) {
            this.updateRange(node * 2, l, mid, ql, qr, delta);
        }

        if (qr > mid) {
            this.updateRange(node * 2 + 1, mid + 1, r, ql, qr, delta);
        }

        this.pushUp(node);
    }

    update(ql, qr, delta) {
        if (ql <= qr) {
            this.updateRange(1, 0, this.n - 1, ql, qr, delta);
        }
    }

    findFirstRange(node, l, r, ql, qr, target) {
        if (r < ql || l > qr) {
            return -1;
        }

        if (target < this.tree[node].min || target > this.tree[node].max) {
            return -1;
        }

        if (l === r) {
            return l;
        }

        this.pushDown(node);

        const mid = Math.floor((l + r) / 2);

        const left = this.findFirstRange(node * 2, l, mid, ql, qr, target);
        if (left !== -1) {
            return left;
        }

        return this.findFirstRange(node * 2 + 1, mid + 1, r, ql, qr, target);
    }

    findFirst(ql, qr, target) {
        if (ql > qr) {
            return -1;
        }

        return this.findFirstRange(1, 0, this.n - 1, ql, qr, target);
    }
}

var longestBalanced = function(nums) {
    const n = nums.length;
    const seg = new LazySegmentTree(n + 1);

    const last = new Map();
    let res = 0;
    let curSum = 0;

    for (let i = 1; i <= n; i++) {
        const x = nums[i - 1];
        const value = x % 2 === 1 ? 1 : -1;

        const prev = last.get(x) || 0;

        if (prev === 0) {
            curSum += value;
            seg.update(i, n, value);
        } else {
            seg.update(prev, i - 1, -value);
        }

        last.set(x, i);

        const j = seg.findFirst(0, i - 1 - res, curSum);
        if (j !== -1) {
            res = i - j;
        }
    }

    return res;
};
```

```python
from typing import List

min = lambda a, b: b if b < a else a
max = lambda a, b: b if b > a else a


class Node:
    __slots__ = 'min', 'max', 'todo'

    def __init__(self):
        self.min = 0
        self.max = 0
        self.todo = 0


class LazySegmentTree:
    def __init__(self, n: int):
        self.n = n
        self.tree = [Node() for _ in range(2 << (n - 1).bit_length())]

    def _apply(self, node: int, delta: int) -> None:
        cur = self.tree[node]
        cur.min += delta
        cur.max += delta
        cur.todo += delta

    def _pushdown(self, node: int) -> None:
        todo = self.tree[node].todo
        if todo == 0:
            return

        self._apply(node * 2, todo)
        self._apply(node * 2 + 1, todo)
        self.tree[node].todo = 0

    def _pushup(self, node: int) -> None:
        left = self.tree[node * 2]
        right = self.tree[node * 2 + 1]

        self.tree[node].min = min(left.min, right.min)
        self.tree[node].max = max(left.max, right.max)

    def _update(self, node: int, l: int, r: int, ql: int, qr: int, delta: int) -> None:
        if ql <= l and r <= qr:
            self._apply(node, delta)
            return

        self._pushdown(node)

        mid = (l + r) // 2
        if ql <= mid:
            self._update(node * 2, l, mid, ql, qr, delta)
        if qr > mid:
            self._update(node * 2 + 1, mid + 1, r, ql, qr, delta)

        self._pushup(node)

    def update(self, ql: int, qr: int, delta: int) -> None:
        if ql <= qr:
            self._update(1, 0, self.n - 1, ql, qr, delta)

    def _find_first(self, node: int, l: int, r: int, ql: int, qr: int, target: int) -> int:
        if r < ql or l > qr:
            return -1

        cur = self.tree[node]
        if target < cur.min or target > cur.max:
            return -1

        if l == r:
            return l

        self._pushdown(node)

        mid = (l + r) // 2

        left_res = self._find_first(node * 2, l, mid, ql, qr, target)
        if left_res != -1:
            return left_res

        return self._find_first(node * 2 + 1, mid + 1, r, ql, qr, target)

    def find_first(self, ql: int, qr: int, target: int) -> int:
        if ql > qr:
            return -1
        return self._find_first(1, 0, self.n - 1, ql, qr, target)


class Solution:
    def longestBalanced(self, nums: List[int]) -> int:
        n = len(nums)
        seg = LazySegmentTree(n + 1)

        last = {}
        res = 0
        cur_sum = 0

        for i, x in enumerate(nums, 1):
            value = 1 if x % 2 else -1
            prev = last.get(x, 0)

            if prev == 0:
                cur_sum += value
                seg.update(i, n, value)
            else:
                seg.update(prev, i - 1, -value)

            last[x] = i

            j = seg.find_first(0, i - 1 - res, cur_sum)
            if j != -1:
                res = i - j

        return res
```

```java
import java.util.*;

class Solution {
    static class Node {
        int min;
        int max;
        int lazy;
    }

    static class LazySegmentTree {
        int n;
        Node[] tree;

        LazySegmentTree(int n) {
            this.n = n;
            tree = new Node[4 * n];
            for (int i = 0; i < tree.length; i++) {
                tree[i] = new Node();
            }
        }

        private void apply(int node, int delta) {
            tree[node].min += delta;
            tree[node].max += delta;
            tree[node].lazy += delta;
        }

        private void pushDown(int node) {
            int lazy = tree[node].lazy;
            if (lazy == 0) {
                return;
            }

            apply(node * 2, lazy);
            apply(node * 2 + 1, lazy);
            tree[node].lazy = 0;
        }

        private void pushUp(int node) {
            tree[node].min = Math.min(tree[node * 2].min, tree[node * 2 + 1].min);
            tree[node].max = Math.max(tree[node * 2].max, tree[node * 2 + 1].max);
        }

        private void update(int node, int l, int r, int ql, int qr, int delta) {
            if (ql <= l && r <= qr) {
                apply(node, delta);
                return;
            }

            pushDown(node);

            int mid = l + (r - l) / 2;
            if (ql <= mid) {
                update(node * 2, l, mid, ql, qr, delta);
            }
            if (qr > mid) {
                update(node * 2 + 1, mid + 1, r, ql, qr, delta);
            }

            pushUp(node);
        }

        void update(int ql, int qr, int delta) {
            if (ql <= qr) {
                update(1, 0, n - 1, ql, qr, delta);
            }
        }

        private int findFirst(int node, int l, int r, int ql, int qr, int target) {
            if (r < ql || l > qr) {
                return -1;
            }

            if (target < tree[node].min || target > tree[node].max) {
                return -1;
            }

            if (l == r) {
                return l;
            }

            pushDown(node);

            int mid = l + (r - l) / 2;
            int left = findFirst(node * 2, l, mid, ql, qr, target);
            if (left != -1) {
                return left;
            }

            return findFirst(node * 2 + 1, mid + 1, r, ql, qr, target);
        }

        int findFirst(int ql, int qr, int target) {
            if (ql > qr) {
                return -1;
            }
            return findFirst(1, 0, n - 1, ql, qr, target);
        }
    }

    public int longestBalanced(int[] nums) {
        int n = nums.length;
        LazySegmentTree seg = new LazySegmentTree(n + 1);

        Map<Integer, Integer> last = new HashMap<>();
        int res = 0;
        int curSum = 0;

        for (int i = 1; i <= n; i++) {
            int x = nums[i - 1];
            int value = (x % 2 == 1) ? 1 : -1;

            int prev = last.getOrDefault(x, 0);

            if (prev == 0) {
                curSum += value;
                seg.update(i, n, value);
            } else {
                seg.update(prev, i - 1, -value);
            }

            last.put(x, i);

            int j = seg.findFirst(0, i - 1 - res, curSum);
            if (j != -1) {
                res = i - j;
            }
        }

        return res;
    }
}
```
