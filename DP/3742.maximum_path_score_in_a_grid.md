# 3742.网格中得分最大的路径

题目链接：[3742.网格中得分最大的路径](https://leetcode.cn/problems/maximum-path-score-in-a-grid/)

## 题目大意

给你一个 m x n 的网格 grid，其中每个单元格包含以下值之一：0、1 或 2。另给你一个整数 k

你从左上角 (0, 0) 出发，目标是到达右下角 (m - 1, n - 1)，只能向 右 或 下 移动

每个单元格根据其值对路径有以下贡献：
- 值为 0 的单元格：分数增加 0，花费 0
- 值为 1 的单元格：分数增加 1，花费 1
- 值为 2 的单元格：分数增加 2，花费 1

返回在总花费不超过 k 的情况下可以获得的 最大分数 ，如果不存在有效路径，则返回 -1

注意： 如果到达最后一个单元格时总花费超过 k，则该路径无效

限制：
- 1 <= m, n <= 200
- 0 <= k <= 10^3​​​​​​​
- ​- ​grid[0][0] == 0
- 0 <= grid[i][j] <= 2

## 解题

```js
/**
 * @param {number[][]} grid
 * @param {number} k
 * @return {number}
 */
var maxPathScore = function(grid, K) {
    const m = grid.length;
    const n = grid[0].length;

    const INF = Number.POSITIVE_INFINITY;
    const minCost = Array(n + 1).fill(INF);
    minCost[1] = 0;

    for (const row of grid) {
        for (let j = 0; j < n; j++) {
            const cellCost = row[j] === 0 ? 0 : 1;
            minCost[j + 1] = Math.min(minCost[j], minCost[j + 1]) + cellCost;
        }
    }

    if (minCost[n] > K) {
        return -1;
    }

    K = Math.min(K, m + n - 2);

    const NEG_INF = Number.NEGATIVE_INFINITY;
    const dp = Array.from({ length: n + 1 }, () => Array(K + 2).fill(NEG_INF));

    dp[1][1] = 0;

    for (let i = 0; i < m; i++) {
        const row = grid[i];

        for (let j = 0; j < n; j++) {
            const x = row[j];
            const cellCost = x === 0 ? 0 : 1;
            const upperCost = Math.min(K, i + j);

            for (let c = upperCost; c >= 0; c--) {
                const prevCost = c - cellCost;

                if (prevCost < 0) {
                    dp[j + 1][c + 1] = NEG_INF;
                    continue;
                }

                const bestPrev = Math.max(
                    dp[j + 1][prevCost + 1],
                    dp[j][prevCost + 1]
                );

                dp[j + 1][c + 1] =
                    bestPrev === NEG_INF ? NEG_INF : bestPrev + x;
            }
        }
    }

    const ans = Math.max(...dp[n]);
    return ans === NEG_INF ? -1 : ans;
};
```

```python
from typing import List
from math import inf

class Solution:
    def maxPathScore(self, grid: List[List[int]], K: int) -> int:
        m, n = len(grid), len(grid[0])

        # Step 1: Check if there exists any path whose cost <= K
        # cost = 1 if cell value > 0, otherwise 0
        min_cost = [inf] * (n + 1)
        min_cost[1] = 0

        for row in grid:
            for j, x in enumerate(row):
                cell_cost = 0 if x == 0 else 1
                min_cost[j + 1] = min(min_cost[j], min_cost[j + 1]) + cell_cost

        # The start cell grid[0][0] is usually guaranteed to be 0.
        # If the minimum required cost is already larger than K, no valid path exists.
        if min_cost[n] > K:
            return -1

        # Step 2: The path length is fixed.
        # From (0,0) to (m-1,n-1), we visit m+n-1 cells.
        # If grid[0][0] = 0, the maximum useful cost is at most m+n-2.
        K = min(K, m + n - 2)

        NEG_INF = float("-inf")

        # dp[j][c] = max score when reaching current row's column j-1
        # using exactly c cost.
        # We use c+1 index to avoid negative index when c-1 appears.
        dp = [[NEG_INF] * (K + 2) for _ in range(n + 1)]

        # Start before (0,0)
        dp[1][1] = 0

        for i, row in enumerate(grid):
            for j, x in enumerate(row):
                cell_cost = 0 if x == 0 else 1

                # From (0,0) to (i,j), the path visits i+j moves after start.
                # So the number of positive-cost cells cannot exceed i+j.
                upper_cost = min(K, i + j)

                # Reverse order because we update dp[j+1] in place.
                for c in range(upper_cost, -1, -1):
                    prev_cost = c - cell_cost

                    if prev_cost < 0:
                        dp[j + 1][c + 1] = NEG_INF
                        continue

                    best_prev = max(
                        dp[j + 1][prev_cost + 1],  # from top
                        dp[j][prev_cost + 1]        # from left
                    )

                    dp[j + 1][c + 1] = best_prev + x if best_prev != NEG_INF else NEG_INF

        ans = max(dp[n])
        return -1 if ans == NEG_INF else ans
```

```java
import java.util.*;

class Solution {
    public int maxPathScore(int[][] grid, int K) {
        int m = grid.length;
        int n = grid[0].length;

        final int INF = 1_000_000_000;
        int[] minCost = new int[n + 1];
        Arrays.fill(minCost, INF);
        minCost[1] = 0;

        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                int cellCost = grid[i][j] == 0 ? 0 : 1;
                minCost[j + 1] = Math.min(minCost[j], minCost[j + 1]) + cellCost;
            }
        }

        if (minCost[n] > K) {
            return -1;
        }

        K = Math.min(K, m + n - 2);

        final int NEG_INF = -1_000_000_000;
        int[][] dp = new int[n + 1][K + 2];

        for (int j = 0; j <= n; j++) {
            Arrays.fill(dp[j], NEG_INF);
        }

        dp[1][1] = 0;

        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                int x = grid[i][j];
                int cellCost = x == 0 ? 0 : 1;
                int upperCost = Math.min(K, i + j);

                for (int c = upperCost; c >= 0; c--) {
                    int prevCost = c - cellCost;

                    if (prevCost < 0) {
                        dp[j + 1][c + 1] = NEG_INF;
                        continue;
                    }

                    int bestPrev = Math.max(
                        dp[j + 1][prevCost + 1],
                        dp[j][prevCost + 1]
                    );

                    dp[j + 1][c + 1] = bestPrev == NEG_INF ? NEG_INF : bestPrev + x;
                }
            }
        }

        int ans = NEG_INF;
        for (int v : dp[n]) {
            ans = Math.max(ans, v);
        }

        return ans == NEG_INF ? -1 : ans;
    }
}
```

- 空间复杂度：`O(n⋅min(k,m+n-2))`
