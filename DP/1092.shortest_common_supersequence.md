# 1092.最短公共超序列

题目链接：[1092.最短公共超序列](https://leetcode.cn/problems/shortest-common-supersequence/)

## 题目大意

给你两个字符串 str1 和 str2，返回同时以 str1 和 str2 作为 子序列 的最短字符串。如果答案不止一个，则可以返回满足条件的 任意一个 答案

如果从字符串 t 中删除一些字符（也可能不删除），可以得到字符串 s ，那么 s 就是 t 的一个子序列

```js
Example 1:
Input: str1 = "abac", str2 = "cab"
Output: "cabac"
Explanation: 
str1 = "abac" is a subsequence of "cabac" because we can delete the first "c".
str2 = "cab" is a subsequence of "cabac" because we can delete the last "ac".
The answer provided is the shortest such string that satisfies these properties.

Example 2:
Input: str1 = "aaaaaaaa", str2 = "aaaaaaaa"
Output: "aaaaaaaa"
```

限制:
- 1 <= str1.length, str2.length <= 1000
- str1 and str2 consist of lowercase English letters.

## 解题

```js
/**
 * @param {string} str1
 * @param {string} str2
 * @return {string}
 */
var shortestCommonSupersequence = function(s, t) {
    const n = s.length;
    const m = t.length;

    const f = Array.from({ length: n + 1 }, () => Array(m + 1).fill(0));

    for (let j = 0; j <= m; j++) {
        f[0][j] = j;
    }

    for (let i = 1; i <= n; i++) {
        f[i][0] = i;
    }

    for (let i = 0; i < n; i++) {
        for (let j = 0; j < m; j++) {
            if (s[i] === t[j]) {
                f[i + 1][j + 1] = f[i][j] + 1;
            } else {
                f[i + 1][j + 1] = Math.min(f[i][j + 1], f[i + 1][j]) + 1;
            }
        }
    }

    const res = [];

    let i = n - 1;
    let j = m - 1;

    while (i >= 0 && j >= 0) {
        if (s[i] === t[j]) {
            res.push(s[i]);
            i--;
            j--;
        } else if (f[i + 1][j + 1] === f[i][j + 1] + 1) {
            res.push(s[i]);
            i--;
        } else {
            res.push(t[j]);
            j--;
        }
    }

    while (i >= 0) {
        res.push(s[i]);
        i--;
    }

    while (j >= 0) {
        res.push(t[j]);
        j--;
    }

    return res.reverse().join("");
};
```

```python
class Solution:
    def shortestCommonSupersequence(self, s: str, t: str) -> str:
        n, m = len(s), len(t)

        f = [[0] * (m + 1) for _ in range(n + 1)]
        f[0] = list(range(m + 1))
        for i in range(1, n + 1):
            f[i][0] = i

        for i, x in enumerate(s):
            for j, y in enumerate(t):
                if x == y:
                    f[i + 1][j + 1] = f[i][j] + 1
                else:
                    f[i + 1][j + 1] = min(f[i][j + 1], f[i + 1][j]) + 1

        res = []
        i, j = n - 1, m - 1

        while i >= 0 and j >= 0:
            if s[i] == t[j]:
                res.append(s[i])
                i -= 1
                j -= 1
            elif f[i + 1][j + 1] == f[i][j + 1] + 1:
                res.append(s[i])
                i -= 1
            else:
                res.append(t[j])
                j -= 1

        return s[:i + 1] + t[:j + 1] + ''.join(reversed(res))
```

```java
class Solution {
    public String shortestCommonSupersequence(String s, String t) {
        int n = s.length();
        int m = t.length();

        int[][] f = new int[n + 1][m + 1];

        for (int j = 0; j <= m; j++) {
            f[0][j] = j;
        }

        for (int i = 1; i <= n; i++) {
            f[i][0] = i;
        }

        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (s.charAt(i) == t.charAt(j)) {
                    f[i + 1][j + 1] = f[i][j] + 1;
                } else {
                    f[i + 1][j + 1] = Math.min(f[i][j + 1], f[i + 1][j]) + 1;
                }
            }
        }

        StringBuilder res = new StringBuilder();

        int i = n - 1;
        int j = m - 1;

        while (i >= 0 && j >= 0) {
            if (s.charAt(i) == t.charAt(j)) {
                res.append(s.charAt(i));
                i--;
                j--;
            } else if (f[i + 1][j + 1] == f[i][j + 1] + 1) {
                res.append(s.charAt(i));
                i--;
            } else {
                res.append(t.charAt(j));
                j--;
            }
        }

        while (i >= 0) {
            res.append(s.charAt(i));
            i--;
        }

        while (j >= 0) {
            res.append(t.charAt(j));
            j--;
        }

        return res.reverse().toString();
    }
}
```
