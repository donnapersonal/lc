# 1594.矩阵的最大非负积

题目链接：[1594.矩阵的最大非负积](https://leetcode.cn/problems/maximum-non-negative-product-in-a-matrix/)

## 题目大意

给你一个大小为 m x n 的矩阵 grid 。最初，你位于左上角 (0, 0) ，每一步，你可以在矩阵中 向右 或 向下 移动

在从左上角 (0, 0) 开始到右下角 (m - 1, n - 1) 结束的所有路径中，找出具有 最大非负积 的路径。路径的积是沿路径访问的单元格中所有整数的乘积

返回 最大非负积 对 109 + 7 取余 的结果。如果最大积为 负数 ，则返回 -1 

注意，取余是在得到最大积之后执行的

![example2](https://github.com/donnapersonal/picx-images-hosting/raw/master/image.45iajt7jlb.webp)

限制：
- m == grid.length, n == grid[i].length
- 1 <= m, n <= 15
- -4 <= grid[i][j] <= 4

## 解题

```js
var maxProductPath = function(grid) {
    const MOD = 1000000007;
    const m = grid.length;
    const n = grid[0].length;

    const fMin = new Array(n).fill(0);
    const fMax = new Array(n).fill(0);

    for (let i = 0; i < m; i++) {
        const row = grid[i];

        for (let j = 0; j < n; j++) {
            const x = row[j];

            if (i === 0 && j === 0) {
                fMin[0] = x;
                fMax[0] = x;
                continue;
            }

            let resMin = Infinity;
            let resMax = -Infinity;

            // From top
            if (i > 0) {
                const mn = fMin[j];
                const mx = fMax[j];

                resMin = Math.min(mn * x, mx * x);
                resMax = Math.max(mn * x, mx * x);
            }

            // From left
            if (j > 0) {
                const mn = fMin[j - 1];
                const mx = fMax[j - 1];

                resMin = Math.min(resMin, mn * x, mx * x);
                resMax = Math.max(resMax, mn * x, mx * x);
            }

            fMin[j] = resMin;
            fMax[j] = resMax;
        }
    }

    const res = fMax[n - 1];
    return res < 0 ? -1 : res % MOD;
};
```

```python
# 2D DP
class Solution:
    def maxProductPath(self, grid: List[List[int]]) -> int:
        m, n = len(grid), len(grid[0])
        f = [[None] * n for _ in range(m)]

        for i, row in enumerate(grid):
            for j, x in enumerate(row):
                if i == j == 0:
                    f[0][0] = (x, x)
                    continue

                res_min, res_max = inf, -inf
                if i > 0:
                    mn, mx = f[i - 1][j]
                    res_min = min(mn * x, mx * x)
                    res_max = max(mn * x, mx * x)
                if j > 0:
                    mn, mx = f[i][j - 1]
                    res_min = min(res_min, mn * x, mx * x)
                    res_max = max(res_max, mn * x, mx * x)

                f[i][j] = (res_min, res_max)

        res = f[-1][-1][1]
        return -1 if res < 0 else res % 1_000_000_007
```

```java
class Solution {
    public int maxProductPath(int[][] grid) {
        int MOD = 1_000_000_007;
        int m = grid.length;
        int n = grid[0].length;

        long[] fMin = new long[n];
        long[] fMax = new long[n];

        for (int i = 0; i < m; i++) {
            int[] row = grid[i];

            for (int j = 0; j < n; j++) {
                long x = row[j];

                if (i == 0 && j == 0) {
                    fMin[0] = x;
                    fMax[0] = x;
                    continue;
                }

                long resMin = Long.MAX_VALUE;
                long resMax = Long.MIN_VALUE;

                // From top
                if (i > 0) {
                    long mn = fMin[j];
                    long mx = fMax[j];

                    resMin = Math.min(mn * x, mx * x);
                    resMax = Math.max(mn * x, mx * x);
                }

                // From left
                if (j > 0) {
                    long mn = fMin[j - 1];
                    long mx = fMax[j - 1];

                    resMin = Math.min(resMin, Math.min(mn * x, mx * x));
                    resMax = Math.max(resMax, Math.max(mn * x, mx * x));
                }

                fMin[j] = resMin;
                fMax[j] = resMax;
            }
        }

        long res = fMax[n - 1];

        if (res < 0) {
            return -1;
        }

        return (int)(res % MOD);
    }
}
```

- 空间复杂度：`O(n)`
